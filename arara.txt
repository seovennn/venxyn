# 1. سیکیورٹی: اہم فائلوں تک رسائی روکیں
<FilesMatch "^(\.htaccess|config\.inc\.php)$">
    <IfModule mod_authz_core.c>
        Require all denied
    </IfModule>
    <IfModule !mod_authz_core.c>
        Order deny,allow
        Deny from all
    </IfModule>
</FilesMatch>

# ڈائریکٹری لسٹنگ بند کریں
Options -Indexes

# 2. پی ایچ پی ہینڈلر (آپ کے سرور کے مطابق)
<IfModule mime_module>
  AddHandler application/x-httpd-ea-php80 .php .php8 .phtml
</IfModule>

# 3. ری رائٹ انجن اور OJS رولز
RewriteEngine On
RewriteBase /

# بلاک کریں اگر کوئی یو آر ایل میں مشکوک سکرپٹ چلانے کی کوشش کرے
RewriteCond %{QUERY_STRING} (<|%3C).*script.*(>|%3E) [NC,OR]
RewriteCond %{QUERY_STRING} GLOBALS(=|\[|\%[0-9A-Z]{0,2}) [OR]
RewriteCond %{QUERY_STRING} _REQUEST(=|\[|\%[0-9A-Z]{0,2})
RewriteRule ^(.*)$ index.php [F,L]

# سائڈ میپ کا فکس (Sitemap Fix)
RewriteRule ^index.php/ojs/sitemap\.xml$ index.php/ojs/sitemap [L,R=301]

# اسٹینڈرڈ OJS رولز
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^(.*)$ index.php/$1 [L]
